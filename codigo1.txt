import java.util.Scanner;
public class ComparadorNum {
    public static void main (String[] args)
    {
        Scanner entrada = new Scanner(System.in);
        System.out.print("Valor numero 1:");
       double num1 =  entrada.nextInt();

       System.out.print("Valor numero 2:");
       double num2 = entrada.nextInt();

       System.out.printf("math max (num1,num2) = %f\n",
       Math.max(num1,num2));

        System.out.printf("math min (num1,num2) = %f\n",
        Math.min(num1,num2));







    }
}
